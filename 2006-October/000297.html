        --prefix=/usr \
        --sysconfdir=/etc \
        --mandir=/usr/share/man \
        --with-sharedir=/usr/share/mutt-ng \
        --with-docdir=/usr/share/doc/mutt-ng \
        --with-mailpath=/var/mail \
        --enable-fcntl --enable-pop \
        --with-sasl2 --with-idn \
        --with-curses \
        --enable-nntp \
        --enable-debug \
        --enable-compressed \
        --with-gnutls \
        --enable-hcache \
        --without-gdbm \
        --enable-gpgme
